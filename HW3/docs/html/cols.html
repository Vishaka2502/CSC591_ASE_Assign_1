from HW3.src.num import NUM
from HW3.src.row import ROW
from HW3.src.sym import SYM
# 
class COLS:
#

Generate NUMs and SYMs from column names

    def __init__(self, t: list) -> None:
# 
        self.names = t
        self.all = []
        self.x = []
        self.y = []
        self.klass = None

        for n, col_name in enumerate(t):
            col = NUM(n, col_name) if col_name[0].isupper() else SYM(n, col_name)
            self.all.append(col)
            if not col_name[-1] == "X":
                if "!" in col_name:
                    self.klass = col
                if "+" in col_name or "-" in col_name or "!" in col_name:
                    self.y.append(col)
                else:
                    self.x.append(col)
#

Updates the (not skipped) columns with details from row :param row: ROW:

    def add(self, row: ROW) -> None:
# 
        for col in self.x + self.y:
            col.add(row.cells[col.at])
